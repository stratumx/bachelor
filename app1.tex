import math
from pprint import pprint

import matplotlib.pyplot as plt
import numpy as np
from mpl_toolkits.mplot3d import Axes3D

# входные данные
LEANING_ANGLE = 0	# угол положения
R = 70		# радиус
ROTCOUNT = 17	# число витков
FINENESS = 0.25		# шаг

# расчёт вертикального и горизонтального углов
v_angle_unit = math.pi / RADIUS / 2
h_angle_unit = math.pi / RADIUS * ROTOUNT * FINENESS

xr = LEANING_ANGLE / 180 * math.pi
xrc = math.cos(xr)
xrs = math.sin(xr)

pnts = []
total_rot = 0
flg = -1

# генерация точек кривой
i = -RADIUS
while i <= RADIUS:
  x = math.cos(i * v_angle_unit) * R
  y = math.sin(i * v_angle_unit) * R
  
  v = [x*math.cos(total_rot), y, x*math.sin(total_rot)]
  
  pnt_y = v[1] * xrc - v[2] * xrs
  pnt_z = v[2] * xrc + v[1] * xrs
  
  pnts.append([v[0], pnt_y, pnt_z])
  total_rot += h_angle_unit
  i += FINENESS

# процедура формирования файла с координатами
n = len(pnts)
coords = open('./line-'+str(R)+'-'+str(ROTCOUNT)+'.sldcrv','w')
for i in range(n):
  xr = round(pnts[i][2],3)
  yr = round(pnts[i][1],3)
  zr = round(pnts[i][0],3)
  coords.write(str(xr)+' '+str(yr)+' '+str(zr)+'\n')
coords.close()

# вывод графика
fig = plt.figure()
ax = fig.gca(projection='3d')

X = np.zeros(n)
Y = np.zeros(n)
Z = np.zeros(n)

for i in range(n):
  for j in range(3):
    if j == 0:
      Z[i] = pnts[i][j]
    elif j == 1:
      Y[i] = pnts[i][j]
    else:
      X[i] = pnts[i][j]

ax.plot(X, Y, Z, label='curve')
 
ax.set_xlabel(r'x')
ax.set_ylabel(r'y')
ax.set_zlabel(r'z')

fig.savefig("Test.svg",bbox_inches="tight",pad_inches=.15)

plt.show()
